Template

for service in services; do;
cat >"stack/${service}.yml" <<EOF

version: '3'
services:
  ${service}-service:
    build: services/${service}
    depends_on:
    - db

EOF
done

#!/bin/bash -e

CONFIG_FILES=(stack/*.yml)

docker-compose ${CONFIG_FILES[@]/#/-f} ${@}
